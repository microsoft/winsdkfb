285  private async Task<FBTestUser> createTestUser(
286  PropertySet Parameters
287  )
288  {
289  FBTestUser user = null;
290  bool success = false;
291 
292  // If you're creating/deleting test users frequently, like say in
293  // a unit test suite, Facebook will occasionally fail this call with
294  // the error code 1, type "OAuthException" and the message "An
295  // unknown error occurred". Throwing in a short delay and retrying
296  // almost always alleviates the problem, so adding a few retries
297  // here greatly increases the robustness of the test suite.
298  for (int retries = 0; (success == false) && (retries < 5);
299  retries++)
300  {
301  string path = "/" + TestAppId + FBSDKTestUsersPath;
302 
303  FBSingleValue sval = new FBSingleValue(path, Parameters,
304  new FBJsonClassFactory(FBTestUser.FromJson));
305 
306  FBResult fbresult = await sval.Post();
307 
308  if ((fbresult.Succeeded == false) || (fbresult.Object == null))
309  {
310  await Task.Delay(TEST_RETRY_DELAY);
311  }
312  else
313  {
314  try
315  {
316  user = (FBTestUser)fbresult.Object;
317  success = true;
318  }
319  catch (InvalidCastException)
320  {
321  Assert.IsTrue(false, "Item returned is not expected type" +
322  " (FBTestUser)");
323  }
324  }
325  }
326 
327  Assert.IsNotNull(user);

329  return user;
